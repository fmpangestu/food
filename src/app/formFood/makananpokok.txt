function getFoodCategoryByName(food: Food): FoodCategory {
  const name = food.name.toLowerCase();

  // Daftar kata kunci
  const laukKeywords = ['ayam', 'ikan', 'sapi', 'telur', 'daging', 'tahu', 'tempe'];
  const pokokKeywords = ['nasi', 'kentang', 'roti', 'mie', 'oatmeal'];
  const sayurKeywords = ['sayur', 'bayam', 'kangkung', 'brokoli', 'tumis', 'capcay'];
  const buahKeywords = ['buah', 'pisang', 'apel', 'jeruk', 'puding', 'yogurt'];

  if (laukKeywords.some(keyword => name.includes(keyword))) {
    return 'lauk_pauk';
  }
  if (pokokKeywords.some(keyword => name.includes(keyword))) {
    return 'makanan_pokok';
  }
  if (sayurKeywords.some(keyword => name.includes(keyword))) {
    return 'sayuran';
  }
  if (buahKeywords.some(keyword => name.includes(keyword))) {
    return 'buah_pencuci_mulut';
  }

  return 'tidak_diketahui';
}
